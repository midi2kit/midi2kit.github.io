Quick Start

Get up and running in minutes

Package.swift 
dependencies: [
    .package(url: "https://github.com/midi2kit/midi2kit-core.git", from: "1.0.0")
]
DeviceDiscovery.swift 
import MIDI2Kit

// Create transport and CI manager
let transport = try CoreMIDITransport(clientName: "MyApp")
let ciManager = CIManager(transport: transport)

// Start discovery
try await ciManager.start()

// Listen for devices
for await event in ciManager.events {
    switch event {
    case .deviceDiscovered(let device):
        print("Found: \(device.displayName)")
        if device.supportsPropertyExchange {
            print("  ‚Üí Supports Property Exchange!")
        }
    case .deviceLost(let muid):
        print("Lost device: \(muid)")
    default:
        break
    }
}
PropertyExchange.swift 
import MIDI2Kit

// Setup PE Manager with destination resolver
let peManager = PEManager(transport: transport, sourceMUID: ciManager.muid)
peManager.destinationResolver = ciManager.makeDestinationResolver()
await peManager.startReceiving()

// Get device info (async/await)
let deviceInfo = try await peManager.getDeviceInfo(from: device.muid)
print("Product: \(deviceInfo.productName ?? "Unknown")")

// Get resource list
let resources = try await peManager.getResourceList(from: device.muid)
for resource in resources {
    print("  \(resource.resource): GET=\(resource.canGet ?? false)")
}

// Subscribe to notifications
let response = try await peManager.subscribe(to: "CurrentProgram", on: device.muid)
for await notification in peManager.startNotificationStream() {
    print("Changed: \(notification.resource)")
}
