DeviceDiscovery.swift 
import MIDI2Kit

// Create client with high-level API
let client = try MIDI2Client(name: "MyApp")
try await client.start()

// Listen for devices
for await event in await client.makeEventStream() {
    switch event {
    case .deviceDiscovered(let device):
        print("Found: \(device.displayName)")
        if device.supportsPropertyExchange {
            // Get device info automatically
            if let info = try? await device.deviceInfo {
                print("  Product: \(info.productName ?? "Unknown")")
            }
        }
    case .deviceLost(let muid):
        print("Lost device: \(muid)")
    default:
        break
    }
}
PropertyExchange.swift 
import MIDI2Kit

// Using MIDI2Device for type-safe property access
if case .deviceDiscovered(let device) = event {
    // Get device info (cached automatically)
    let info = try await device.deviceInfo
    print("Product: \(info.productName ?? "Unknown")")

    // Get resource list
    let resources = try await device.resourceList
    for resource in resources {
        print("  \(resource.resource): GET=\(resource.canGet ?? false)")
    }

    // Type-safe property access
    struct VolumeInfo: Codable { let level: Int }
    if let volume = try await device.getProperty("Volume", as: VolumeInfo.self) {
        print("Volume: \(volume.level)")
    }
}
MockDeviceTest.swift 
import MIDI2Kit
import MIDI2Transport
import MIDI2CI
import MIDI2PE

// Create loopback transport pair for in-process testing
let (initiatorTransport, responderTransport) = LoopbackTransport.createPair()

// Create MockDevice with KORG preset
let mockDevice = MockDevice(
    transport: responderTransport,
    preset: .korgModulePro
)
try await mockDevice.start()

// Create initiator-side managers
let ciManager = CIManager(transport: initiatorTransport, ...)
let peManager = PEManager(transport: initiatorTransport, ...)
try await ciManager.start()

// Discover the mock device
await ciManager.startDiscovery()
// MockDevice responds to Discovery Inquiry automatically

// PE requests work just like with real devices
let response = try await peManager.get("DeviceInfo", from: mockDevice.handle)
print("Mock device info: \(response.bodyString)")
